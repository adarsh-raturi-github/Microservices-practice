// Creating posts image
cd posts
docker build -t adarsh893/posts:0.0.1 .
cd k8s
kubectl apply -f posts.yaml

// Now create deployement post-depl.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: posts-depl
spec:
  replicas: 1
  selector:
    matchLabels:
      app: posts
  template:
    metadata:
      labels:
        app: posts
    spec:
      containers:
        - name: posts-container
          image: adarsh893/posts:0.0.1

# apiVersion → Which API handles this resource.
# kind → Type of resource (Deployment).
# metadata → Information about the Deployment.
# name → Deployment name.
# spec → Desired state of the Deployment.
# replicas → Number of Pods to maintain.
# selector → Which Pods this Deployment manages.
# matchLabels → Labels used to identify Pods.
# template → Blueprint for creating Pods.
# template.metadata → Metadata for created Pods.
# labels → Tags attached to Pods.
# template.spec → Defines what Pods run.
# containers → Containers inside the Pod.
# container name → Name of the container.
# image → Docker image to execute.
